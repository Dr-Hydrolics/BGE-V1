348  float dist = glm::length(intersection);

351  dist = distToClosestIP;

354  }
355  }
356  }
357  }
358  if (closestIntersectingObstacle != nullptr)
359  {
360  // Now calculate the force
361  // Calculate Z Axis braking force
362  float multiplier = 200 * (1.0f + (boxLength - localPosOfClosestObstacle.z) / boxLength);
363 
364  //calculate the lateral force
365  float expandedRadius = transform->scale.x + closestIntersectingObstacle->transform->scale.x;
366  force.x = (expandedRadius - glm::abs(localPosOfClosestObstacle.x)) * multiplier;
367  force.y = (expandedRadius - -glm::abs(localPosOfClosestObstacle.y)) * multiplier;
368

371  force.x = -force.x;
372  }
373 
374  if (localPosOfClosestObstacle.y > 0)

376  force.y = -force.y;
377  }
378  //apply a braking force proportional to the obstacle's distance from
379  //the vehicle.
380  const float brakingWeight = 40.0f;
381  force.z = (closestIntersectingObstacle->transform->scale.x -
382  localPosOfClosestObstacle.z) *
383  brakingWeight;
384 
385  //finally, convert the steering vector from local to world space
386  force = glm::vec3(transform->world * glm::vec4(force, 1.0f));
387  }
388  }
